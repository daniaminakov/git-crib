dir - отобразить список папок и файлов.
clear - очистить консоль.
cd  - перейти.
mkdir - создать папку.

git config --global user.name - имя
git config --global user.email - email
git init - иницилизация репозитория.
git status - статус.
git add название_файла - добавить на отслеживания файл.
git add . - добавить на отслеживания все файлы.
git restore - отменяет изменения, внесенные другими коммитами.
git commit -m "" - коммит с message(название коммита).
git log - показывает историю всех коммитов в данной ветке с их хэшами и сообщениями.
git branch - список веток.
git branch название_ветки - создать ветку.
git branch -D название_ветки - удалить ветку.
git checkout название_хэша_коммита - переключится на предыдущие версии(коммиты).
git checkout название_ветки - переключится на ветку.
git checkout -b название_ветки - создать ветку и сразу переключиться на нее.
git merge название_ветки - Слияние веток.
git push - запушить на github.
git clone урл_репозитория - склонировать с github.
git pull - забрать все последние изминения которые были(перед тем, как начать работу).
git pull request - запрос на вливание изменений из вашей ветки в основную ветку.

git branch -M main - переименовать ветку "master" в "main" git

Привязка удаленного репозитория к локальному
git remote add origin https://github.com/daniaminakov/git-crib.git
git push -u origin master
